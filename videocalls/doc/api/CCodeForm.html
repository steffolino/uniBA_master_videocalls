public function run()
{
    $templates=array();
    foreach($this->model->getTemplates() as $i=>$template)
        $templates[$i]=basename($template).' ('.$template.')';

    $this->renderFile(Yii::getPathOfAlias('gii.views.common.generator').'.php',array(
        'model'=>$this->model,
        'templates'=>$templates,
    ));

    parent::run();

    echo "</div>";
}
